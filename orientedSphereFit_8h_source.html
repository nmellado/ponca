177  PONCA_MULTIARCH inline ScalarArray dprattNorm2() const
178  {
179  return Scalar(2.) * Base::m_ul.transpose() * m_dUl
180  - Scalar(4.) * Base::m_uq * m_dUc
181  - Scalar(4.) * Base::m_uc * m_dUq;
182  }
183 
185  PONCA_MULTIARCH inline Scalar dprattNorm2(unsigned int _d) const
186  {
187  return Scalar(2.) * m_dUl.col(_d).dot(Base::m_ul)
188  - Scalar(4.) * m_dUc.col(_d)[0]*Base::m_uq
189  - Scalar(4.) * m_dUq.col(_d)[0]*Base::m_uc;
190  }

194  {
