37 template <class DataPoint, class WeightKernel>
38 typename DistWeightFunc<DataPoint, WeightKernel>::MatrixType
39 DistWeightFunc<DataPoint, WeightKernel>::spaced2w( const VectorType& _q,
40  const DataPoint&) const
41 {
42  MatrixType result = MatrixType::Zero();
43  VectorType q = convertToLocalBasis(_q);
44  Scalar d = q.norm();
45  if (d <= m_t && d != Scalar(0.))
46  {
47  Scalar der = m_wk.df(d/m_t);
48  result = q*q.transpose()/d*(m_wk.ddf(d/m_t)/m_t - der/d);
49  result.diagonal().array() += der;
50  result *= Scalar(1.)/(m_t*d);
51  }
52  return result;
53 }
