174  /* Processing */
175  /************************************************************************/
178  PONCA_MULTIARCH inline bool addLocalNeighbor(Scalar w,
179  const VectorType &localQ,
180  const DataPoint &attributes,
181  ScalarArray &dw)
182  {
183  if( Base::addLocalNeighbor(w, localQ, attributes) ) {
184  int spaceId = (Type & FitScaleDer) ? 1 : 0;
185  // compute weight
186  if (Type & FitScaleDer)
187  dw[0] = Base::m_w.scaledw(attributes.pos(), attributes);
188 
189  if (Type & FitSpaceDer)
190  dw.template segment<int(DataPoint::Dim)>(spaceId) = -Base::m_w.spacedw(attributes.pos(), attributes).transpose();
191 
192  m_dSumW += dw;
193  return true;
194  }
195  return false;
196  }
