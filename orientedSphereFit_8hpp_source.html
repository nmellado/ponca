26 template < class DataPoint, class _WFunctor, typename T>
27 bool
28 OrientedSphereFit<DataPoint, _WFunctor, T>::addNeighbor(const DataPoint& _nei)
29 {
30  // centered basis
31  VectorType q = _nei.pos() - Base::basisCenter();
32 
33  // compute weight
34  Scalar w = m_w.w(q, _nei);
35 
36  if (w > Scalar(0.))
37  {
38  // increment matrix
39  m_sumP += q * w;
40  m_sumN += _nei.normal() * w;
41  m_sumDotPN += w * _nei.normal().dot(q);
42  m_sumDotPP += w * q.squaredNorm();
43  m_sumW += w;
44 
46  ++(Base::m_nbNeighbors);
47  return true;
48  }
49 
50  return false;
51 }

54 template < class DataPoint, class _WFunctor, typename T>
55 FIT_RESULT
56 OrientedSphereFit<DataPoint, _WFunctor, T>::finalize ()
57 {
58  PONCA_MULTIARCH_STD_MATH(sqrt);
59  PONCA_MULTIARCH_STD_MATH(max);
60  PONCA_MULTIARCH_STD_MATH(abs);
61 
62  // 1. finalize sphere fitting
63  Scalar epsilon = Eigen::NumTraits<Scalar>::dummy_precision();
64 
65  // handle specific configurations
66  // With less than 3 neighbors the fitting is undefined
67  if(m_sumW == Scalar(0.) || Base::m_nbNeighbors < 3)
68  {
69  Base::m_ul.setZero();
70  Base::m_uc = Scalar(0.);
71  Base::m_uq = Scalar(0.);
72  Base::m_isNormalized = false;
73  Base::m_eCurrentState = UNDEFINED;
74  return Base::m_eCurrentState;
75  }
76 
77  Scalar invSumW = Scalar(1.)/m_sumW;
78 
79  m_nume = (m_sumDotPN - invSumW * m_sumP.dot(m_sumN));
80  Scalar den1 = invSumW * m_sumP.dot(m_sumP);
81  m_deno = m_sumDotPP - den1;
82 
83  // Deal with degenerate cases
84  if(abs(m_deno) < epsilon * max(m_sumDotPP, den1))
85  {
86  //plane
87  Scalar s = Scalar(1.) / Base::m_ul.norm();
88  Base::m_ul = s*Base::m_ul;
89  Base::m_uc = s*Base::m_uc;
90  Base::m_uq = Scalar(0.);
91  }
92  else
93  {
94  //Generic case
95  Base::m_uq = Scalar(.5) * m_nume / m_deno;
96  Base::m_ul = (m_sumN - m_sumP * (Scalar(2.) * Base::m_uq)) * invSumW;
97  Base::m_uc = -invSumW * (Base::m_ul.dot(m_sumP) + m_sumDotPP * Base::m_uq);
98  }
99 
100  Base::m_isNormalized = false;
101 
102  if(Base::m_nbNeighbors < 6)
103  {
104  Base::m_eCurrentState = UNSTABLE;
105  }
106  else
107  {
108  Base::m_eCurrentState = STABLE;
109  }
110 
111  return Base::m_eCurrentState;
112 }

139 template < class DataPoint, class _WFunctor, typename T, int Type>
140 bool
141 OrientedSphereDer<DataPoint, _WFunctor, T, Type>::addNeighbor(const DataPoint &_nei)
142 {
143  bool bResult = Base::addNeighbor(_nei);
144 
145  if(bResult)
146  {
147  ScalarArray dw;
148 
149  // centered basis
150  VectorType q = _nei.pos() - Base::basisCenter();
151 
152  // compute weight derivatives
153  if (Type & FitScaleDer)
154  dw[0] = Base::m_w.scaledw(q, _nei);
155 
156  if (Type & FitSpaceDer)
157  dw.template tail<int(DataPoint::Dim)>() = -Base::m_w.spacedw(q, _nei).transpose();
158 
159  // increment
160  m_dSumW += dw;
161  m_dSumP += q * dw;
162  m_dSumN += _nei.normal() * dw;
163  m_dSumDotPN += dw * _nei.normal().dot(q);
164  m_dSumDotPP += dw * q.squaredNorm();
165 
166  return true;
167  }

169  return false;

173 template < class DataPoint, class _WFunctor, typename T, int Type>
174 FIT_RESULT
175 OrientedSphereDer<DataPoint, _WFunctor, T, Type>::finalize()
176 {
177  PONCA_MULTIARCH_STD_MATH(sqrt);
178 
179  Base::finalize();
180  // Test if base finalize end on a viable case (stable / unstable)
181  if (this->isReady())
182  {
183  Scalar invSumW = Scalar(1.)/Base::m_sumW;
184 
185  Scalar nume = Base::m_sumDotPN - invSumW*Base::m_sumP.dot(Base::m_sumN);
186  Scalar deno = Base::m_sumDotPP - invSumW*Base::m_sumP.dot(Base::m_sumP);
187 
188  // FIXME, the following product "Base::m_sumN.transpose() * m_dSumP" is prone to numerical cancellation
189  // issues for spacial derivatives because, (d sum w_i P_i)/(d x) is supposed to be tangent to the surface whereas
190  // "sum w_i N_i" is normal to the surface...
191  m_dNume = m_dSumDotPN
192  - invSumW*invSumW * ( Base::m_sumW * ( Base::m_sumN.transpose() * m_dSumP + Base::m_sumP.transpose() * m_dSumN )
193  - m_dSumW*Base::m_sumP.dot(Base::m_sumN) );
194 
195  m_dDeno = m_dSumDotPP
196  - invSumW*invSumW*( Scalar(2.) * Base::m_sumW * Base::m_sumP.transpose() * m_dSumP
197  - m_dSumW*Base::m_sumP.dot(Base::m_sumP) );
198 
199  m_dUq = Scalar(.5) * (deno * m_dNume - m_dDeno * nume)/(deno*deno);
200 
201  // FIXME: this line is prone to numerical cancellation issues because dSumN and u_l*dSumW are close to each other.
202  // If using two passes, one could directly compute sum( dw_i + (n_i - ul) ) to avoid this issue.
203  m_dUl = invSumW * ( m_dSumN - Base::m_ul*m_dSumW - Scalar(2.)*(m_dSumP*Base::m_uq + Base::m_sumP*m_dUq) );
204  m_dUc = -invSumW*( Base::m_sumP.transpose() * m_dUl
205  + Base::m_sumDotPP * m_dUq
206  + Base::m_ul.transpose() * m_dSumP
207  + Base::m_uq * m_dSumDotPP
208  + m_dSumW * Base::m_uc);
209  }
210 
211  return Base::m_eCurrentState;
212 }

214 template < class DataPoint, class _WFunctor, typename T, int Type>
215 typename OrientedSphereDer <DataPoint, _WFunctor, T, Type>::VectorArray
216 OrientedSphereDer<DataPoint, _WFunctor, T, Type>::dNormal() const
217 {
218  // Computes the derivatives of the normal of the sphere at the evaluation point.
219  // Therefore, we must take into account the variation of the evaluation point when differentiating wrt space
220  // i.e., normal(x) = grad / |grad|, with grad(x) = ul + 2 uq * x, and diff_x(grad) = dul + 2 uq I
221  VectorArray dgrad = m_dUl;
222  if(isSpaceDer())
223  dgrad.template rightCols<DataPoint::Dim>().diagonal().array() += Scalar(2)*Base::m_uq;
224  Scalar norm = Base::m_ul.norm();
225  Scalar norm3 = norm*norm*norm;
226  return dgrad / norm - Base::m_ul * (Base::m_ul.transpose() * dgrad) / norm3;
227 }

239 template < class DataPoint, class _WFunctor, typename T, int Type>
240 bool
241 OrientedSphereDer<DataPoint, _WFunctor, T, Type>::applyPrattNorm()
242 {
243  if(Base::isNormalized())
244  return false; //need original parameters without Pratt Normalization
245 
246  PONCA_MULTIARCH_STD_MATH(sqrt);
247  Scalar pn2 = Base::prattNorm2();
248  Scalar pn = sqrt(pn2);
249 
250  ScalarArray dpn2 = dprattNorm2();
251  ScalarArray factor = Scalar(0.5) * dpn2 / pn;
252 
253  m_dUc = ( m_dUc * pn - Base::m_uc * factor ) / pn2;
254  m_dUl = ( m_dUl * pn - Base::m_ul * factor ) / pn2;
255  m_dUq = ( m_dUq * pn - Base::m_uq * factor ) / pn2;
256 
257  Base::applyPrattNorm();
258  return true;
259 }
