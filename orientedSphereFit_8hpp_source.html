35 template < class DataPoint, class _WFunctor, typename T>
36 FIT_RESULT
37 OrientedSphereFitImpl<DataPoint, _WFunctor, T>::finalize ()
38 {
39  PONCA_MULTIARCH_STD_MATH(sqrt);
40  PONCA_MULTIARCH_STD_MATH(max);
41  PONCA_MULTIARCH_STD_MATH(abs);
42 
43  // Compute status
44  if(Base::finalize() != STABLE || Base::m_nbNeighbors < 3)
45  return Base::m_eCurrentState;
46  if (Base::algebraicSphere().isValid())
47  Base::m_eCurrentState = CONFLICT_ERROR_FOUND;
48  else
49  Base::m_eCurrentState = Base::m_nbNeighbors < 6 ? UNSTABLE : STABLE;
50 
51  // 1. finalize sphere fitting
52  Scalar epsilon = Eigen::NumTraits<Scalar>::dummy_precision();
53 
54  Scalar invSumW = Scalar(1.)/Base::m_sumW;
55 
56  m_nume = (m_sumDotPN - invSumW * Base::m_sumP.dot(Base::m_sumN));
57  Scalar den1 = invSumW * Base::m_sumP.dot(Base::m_sumP);
58  m_deno = m_sumDotPP - den1;
59 
60  // Deal with degenerate cases
61  if(abs(m_deno) < epsilon * max(m_sumDotPP, den1))
62  {
63  //plane
64  Scalar s = Scalar(1.) / Base::m_ul.norm();
65  Base::m_ul = s*Base::m_ul;
66  Base::m_uc = s*Base::m_uc;
67  Base::m_uq = Scalar(0.);
68  }
69  else
70  {
71  //Generic case
72  Base::m_uq = Scalar(.5) * m_nume / m_deno;
73  Base::m_ul = (Base::m_sumN - Base::m_sumP * (Scalar(2.) * Base::m_uq)) * invSumW;
74  Base::m_uc = -invSumW * (Base::m_ul.dot(Base::m_sumP) + m_sumDotPP * Base::m_uq);
75  }
76 
77  Base::m_isNormalized = false;
78 
79  return Base::m_eCurrentState;
80 }

119 template < class DataPoint, class _WFunctor, int DiffType, typename T>
120 FIT_RESULT
121 OrientedSphereDerImpl<DataPoint, _WFunctor, DiffType, T>::finalize()
122 {
123  PONCA_MULTIARCH_STD_MATH(sqrt);
124 
125  Base::finalize();
126  // Test if base finalize end on a viable case (stable / unstable)
127  if (this->isReady())
128  {
129  Scalar invSumW = Scalar(1.)/Base::m_sumW;
130 
131  Scalar nume = Base::m_sumDotPN - invSumW*Base::m_sumP.dot(Base::m_sumN);
132  Scalar deno = Base::m_sumDotPP - invSumW*Base::m_sumP.dot(Base::m_sumP);
133 
134  // FIXME, the following product "Base::m_sumN.transpose() * m_dSumP" is prone to numerical cancellation
135  // issues for spacial derivatives because, (d sum w_i P_i)/(d x) is supposed to be tangent to the surface whereas
136  // "sum w_i N_i" is normal to the surface...
137  m_dNume = m_dSumDotPN
138  - invSumW*invSumW * ( Base::m_sumW * ( Base::m_sumN.transpose() * Base::m_dSumP + Base::m_sumP.transpose() * m_dSumN )
139  - Base::m_dSumW*Base::m_sumP.dot(Base::m_sumN) );
140 
141  m_dDeno = m_dSumDotPP
142  - invSumW*invSumW*( Scalar(2.) * Base::m_sumW * Base::m_sumP.transpose() * Base::m_dSumP
143  - Base::m_dSumW*Base::m_sumP.dot(Base::m_sumP) );
144 
145  m_dUq = Scalar(.5) * (deno * m_dNume - m_dDeno * nume)/(deno*deno);
146 
147  // FIXME: this line is prone to numerical cancellation issues because dSumN and u_l*dSumW are close to each other.
148  // If using two passes, one could directly compute sum( dw_i + (n_i - ul) ) to avoid this issue.
149  m_dUl = invSumW * ( m_dSumN - Base::m_ul*Base::m_dSumW - Scalar(2.)*(Base::m_dSumP * Base::m_uq + Base::m_sumP * m_dUq) );
150  m_dUc = -invSumW*( Base::m_sumP.transpose() * m_dUl
151  + Base::m_sumDotPP * m_dUq
152  + Base::m_ul.transpose() * Base::m_dSumP
153  + Base::m_uq * m_dSumDotPP
154  + Base::m_dSumW * Base::m_uc);
155  }
156 
157  return Base::m_eCurrentState;
158 }

160 template < class DataPoint, class _WFunctor, int DiffType, typename T>
161 typename OrientedSphereDerImpl <DataPoint, _WFunctor, DiffType, T>::VectorArray
162 OrientedSphereDerImpl<DataPoint, _WFunctor, DiffType, T>::dNormal() const
163 {
164  // Computes the derivatives of the normal of the sphere at the evaluation point.
165  // Therefore, we must take into account the variation of the evaluation point when differentiating wrt space
166  // i.e., normal(x) = grad / |grad|, with grad(x) = ul + 2 uq * x, and diff_x(grad) = dul + 2 uq I
167  VectorArray dgrad = m_dUl;
168  if(Base::isSpaceDer())
169  dgrad.template rightCols<DataPoint::Dim>().diagonal().array() += Scalar(2)*Base::m_uq;
170  Scalar norm = Base::m_ul.norm();
171  Scalar norm3 = norm*norm*norm;
172  return dgrad / norm - Base::m_ul * (Base::m_ul.transpose() * dgrad) / norm3;
173 }

185 template < class DataPoint, class _WFunctor, int DiffType, typename T>
186 bool
187 OrientedSphereDerImpl<DataPoint, _WFunctor, DiffType, T>::applyPrattNorm()
188 {
189  if(Base::isNormalized())
190  return false; //need original parameters without Pratt Normalization
191 
192  PONCA_MULTIARCH_STD_MATH(sqrt);
193  Scalar pn2 = Base::prattNorm2();
194  Scalar pn = sqrt(pn2);
195 
196  ScalarArray dpn2 = dprattNorm2();
197  ScalarArray factor = Scalar(0.5) * dpn2 / pn;
198 
199  m_dUc = ( m_dUc * pn - Base::m_uc * factor ) / pn2;
200  m_dUl = ( m_dUl * pn - Base::m_ul * factor ) / pn2;
201  m_dUq = ( m_dUq * pn - Base::m_uq * factor ) / pn2;
202 
203  Base::applyPrattNorm();
204  return true;
205 }
