165  PONCA_MULTIARCH inline ScalarArray dprattNorm2() const
166  {
167  return Scalar(2.) * Base::m_ul.transpose() * m_dUl
168  - Scalar(4.) * Base::m_uq * m_dUc
169  - Scalar(4.) * Base::m_uc * m_dUq;

173  PONCA_MULTIARCH inline Scalar dprattNorm2(unsigned int _d) const
174  {
175  return Scalar(2.) * m_dUl.col(_d).dot(Base::m_ul)
176  - Scalar(4.) * m_dUc.col(_d)[0]*Base::m_uq
177  - Scalar(4.) * m_dUq.col(_d)[0]*Base::m_uc;
178  }
