8 template < class DataPoint, class _WFunctor, typename T>
9 typename GLSDer <DataPoint, _WFunctor, T>::ScalarArray
10  GLSDer <DataPoint, _WFunctor, T>::dtau() const
11 {
12  PONCA_MULTIARCH_STD_MATH(sqrt);
13 
14  Scalar prattNorm2 = Base::prattNorm2();
15  Scalar prattNorm = sqrt(prattNorm2);
16  Scalar cfactor = Scalar(.5) / prattNorm;
17  ScalarArray dfield = Base::m_dUc;
18  // Recall that tau is the field function at the evaluation point, we thus must take care about
19  // its variation when differentiating in space:
20  if(this->isScaleDer())
21  dfield.template tail<DataPoint::Dim>() += Base::m_ul;
22 
23  return (dfield * prattNorm - Base::m_uc * cfactor * Base::dprattNorm2()) / prattNorm2;
24 }

36 typename GLSDer <DataPoint, _WFunctor, T>::ScalarArray
37 GLSDer <DataPoint, _WFunctor, T>::dkappa() const
38 {

40 
41  Scalar prattNorm2 = Base::prattNorm2();
42  Scalar prattNorm = sqrt(prattNorm2);
43  Scalar cfactor = Scalar(.5) / prattNorm;
