25 template < class DataPoint, class _WFunctor, typename T>
26 bool
27 CovariancePlaneFit<DataPoint, _WFunctor, T>::addNeighbor(const DataPoint& _nei)
28 {
29  VectorType q = _nei.pos() - Base::basisCenter();
30  // compute weight
31  Scalar w = m_w.w(q, _nei);
32 
33  if (w > Scalar(0.))
34  {
35  m_cog += w * q;
36  m_sumW += w;
37  m_cov += w * q * q.transpose();
38 
39  ++(Base::m_nbNeighbors);
40  return true;
41  }
42  return false;
43 }

46 template < class DataPoint, class _WFunctor, typename T>
47 FIT_RESULT
48 CovariancePlaneFit<DataPoint, _WFunctor, T>::finalize ()
49 {
50  // handle specific configurations
51  // With less than 3 neighbors the fitting is undefined
52  if(m_sumW == Scalar(0.) || Base::m_nbNeighbors < 3)
53  {
54  Base::resetPrimitive();
55  Base::m_eCurrentState = UNDEFINED;
56  return Base::m_eCurrentState;
57  }
58 
59  // Finalize the centroid (still expressed in local basis)
60  m_cog = m_cog/m_sumW;
61 
62  // Center the covariance on the centroid
63  m_cov = m_cov/m_sumW - m_cog * m_cog.transpose();
64 
65 #ifdef __CUDACC__
66  m_solver.computeDirect(m_cov);
67 #else
68  m_solver.compute(m_cov);
69 #endif
70  Base::m_eCurrentState = ( m_solver.info() == Eigen::Success ? STABLE : UNDEFINED );
71 
72  Base::setPlane(m_solver.eigenvectors().col(0), m_cog);
73 
74  return Base::m_eCurrentState;
75 }

128 template < class DataPoint, class _WFunctor, typename T, int Type>
129 bool
130 CovariancePlaneDer<DataPoint, _WFunctor, T, Type>::addNeighbor(const DataPoint &_nei)
131 {
132  bool bResult = Base::addNeighbor(_nei);
133 
134  if(bResult)
135  {
136  int spaceId = (Type & FitScaleDer) ? 1 : 0;
137 
138  ScalarArray dw;
139 
140  // centered basis
141  VectorType q = _nei.pos()-Base::basisCenter();
142 
143  // compute weight
144  if (Type & FitScaleDer)
145  dw[0] = Base::m_w.scaledw(q, _nei);
146 
147  if (Type & FitSpaceDer)
148  dw.template segment<int(DataPoint::Dim)>(spaceId) = -Base::m_w.spacedw(q, _nei).transpose();
149 
150  m_dSumW += dw;
151  m_dCog += q * dw;
152  for(int k=0; k<NbDerivatives; ++k)
153  m_dCov[k] += dw[k] * q * q.transpose();
154 
155  return true;
156  }
157 
158  return false;
159 }

162 template < class DataPoint, class _WFunctor, typename T, int Type>
163 FIT_RESULT
164 CovariancePlaneDer<DataPoint, _WFunctor, T, Type>::finalize()
165 {
166  PONCA_MULTIARCH_STD_MATH(sqrt);
167 
168  Base::finalize();
169  // Test if base finalize end on a viable case (stable / unstable)
170  if (this->isReady())
171  {
172  // pre-compute shifted eigenvalues to apply the pseudo inverse of C - lambda_0 I
173  Scalar epsilon = Scalar(2) * Eigen::NumTraits<Scalar>::epsilon();
174  Scalar consider_as_zero = Scalar(2) * std::numeric_limits<Scalar>::denorm_min();
175  Eigen::Matrix<Scalar,2,1> shifted_eivals = Base::m_solver.eigenvalues().template tail<2>().array() - Base::m_solver.eigenvalues()(0);
176  if(shifted_eivals(0) < consider_as_zero || shifted_eivals(0) < epsilon * shifted_eivals(1)) shifted_eivals(0) = 0;
177  if(shifted_eivals(1) < consider_as_zero) shifted_eivals(1) = 0;
178 
179  for(int k=0; k<NbDerivatives; ++k)
180  {
181  // Finalize the computation of dCov.
182  m_dCov[k] = m_dCov[k]
183  - Base::m_cog * m_dCog.col(k).transpose()
184  - m_dCog.col(k) * Base::m_cog.transpose()
185  + m_dSumW[k] * Base::m_cog * Base::m_cog.transpose();
186 
187  // apply normalization by sumW:
188  m_dCog.col(k) = (m_dCog.col(k) - m_dSumW(k) * Base::m_cog) / Base::m_sumW;
189 
190  VectorType normal = Base::primitiveGradient();
191  // The derivative of 'normal' is the derivative of the smallest eigenvector.
192  // Since the covariance matrix is real and symmetric, it is equal to:
193  // n' = - (C - lambda_0 I)^+ C' n
194  // Where ^+ denotes the pseudo-inverse.
195  // Since we already performed the eigenvalue decomposition of the matrix C,
196  // we can directly apply the pseudo inverse by observing that:
197  // (C - lambda_0 I) = V (L - lambda_0 I) V^T
198  // where V is the eigenvector matrix, and L the eigenvalue diagonal matrix.
199  Eigen::Matrix<Scalar,2,1> z = - Base::m_solver.eigenvectors().template rightCols<2>().transpose() * (m_dCov[k] * normal);
200  if(shifted_eivals(0)>0) z(0) /= shifted_eivals(0);
201  if(shifted_eivals(1)>0) z(1) /= shifted_eivals(1);
202  m_dNormal.col(k) = Base::m_solver.eigenvectors().template rightCols<2>() * z;
203 
204  VectorType dDiff = -m_dCog.col(k);
205  if(k>0 || !isScaleDer())
206  dDiff(isScaleDer() ? k-1 : k) += 1;
207  m_dDist(k) = m_dNormal.col(k).dot(Base::m_cog) + normal.dot(dDiff);
208 
209  // \fixme we shouldn't need this normalization, however currently the derivatives are overestimated by a factor 2
210  m_dNormal /= Scalar(2.);
211  }
212  }
213 
214  return Base::m_eCurrentState;
215 }
