26 template < class DataPoint, class _WFunctor, typename T>
27 bool
28 MlsSphereFitDer<DataPoint, _WFunctor, T>::addNeighbor(const DataPoint& _nei)
29 {
30  bool bResult = Base::addNeighbor(_nei);
31 
32  if(bResult)
33  {
34  // centered basis
35  VectorType q = _nei.pos() - Base::basisCenter();
36 
37  // compute weight derivatives
38  Matrix d2w = Matrix::Zero();
39 
40  if (Base::isScaleDer())
41  d2w(0,0) = Base::m_w.scaled2w(q, _nei);
42 
43  if (Base::isSpaceDer())
44  d2w.template bottomRightCorner<Dim,Dim>() = Base::m_w.spaced2w(q, _nei);
45 
46  if (Base::isScaleDer() && Base::isSpaceDer())
47  {
48  d2w.template bottomLeftCorner<Dim,1>() = Base::m_w.scaleSpaced2w(q,_nei);
49  d2w.template topRightCorner<1,Dim>() = d2w.template bottomLeftCorner<Dim,1>().transpose();
50  }
51 
52  m_d2SumDotPN += d2w * _nei.normal().dot(q);
53  m_d2SumDotPP += d2w * q.squaredNorm();
54  m_d2SumW += d2w;
55 
56  for(int i=0; i<Dim; ++i)
57  {
58  m_d2SumP.template block<DerDim,DerDim>(0,i*DerDim) += d2w * q[i];
59  m_d2SumN.template block<DerDim,DerDim>(0,i*DerDim) += d2w * _nei.normal()[i];
60  }
61  }
62  return bResult;
63 }

65 template < class DataPoint, class _WFunctor, typename T>
66 FIT_RESULT
67 MlsSphereFitDer<DataPoint, _WFunctor, T>::finalize()
68 {
69  Base::finalize();
70 
71  if (this->isReady())
72  {
73  Matrix sumdSumPdSumN = Matrix::Zero();
74  Matrix sumd2SumPdSumN = Matrix::Zero();
75  Matrix sumd2SumNdSumP = Matrix::Zero();
76  Matrix sumdSumPdSumP = Matrix::Zero();
77  Matrix sumd2SumPdSumP = Matrix::Zero();
78 
79  for(int i=0; i<Dim; ++i)
80  {
81  sumdSumPdSumN += Base::m_dSumN.row(i).transpose()*Base::m_dSumP.row(i);
82  sumd2SumPdSumN += m_d2SumP.template block<DerDim,DerDim>(0,i*DerDim)*Base::m_sumN(i);
83  sumd2SumNdSumP += m_d2SumN.template block<DerDim,DerDim>(0,i*DerDim)*Base::m_sumP(i);
84  sumdSumPdSumP += Base::m_dSumP.row(i).transpose()*Base::m_dSumP.row(i);
85  sumd2SumPdSumP += m_d2SumP.template block<DerDim,DerDim>(0,i*DerDim)*Base::m_sumP(i);
86  }
87 
88  Scalar invSumW = Scalar(1.)/Base::m_sumW;
89 
90  Matrix d2Nume = m_d2SumDotPN
91  - invSumW*invSumW*invSumW*invSumW*(
92  Base::m_sumW*Base::m_sumW*( Base::m_sumW*(sumdSumPdSumN+sumdSumPdSumN.transpose()+sumd2SumPdSumN+sumd2SumNdSumP)
93  + Base::m_dSumW.transpose()*(Base::m_sumN.transpose()*Base::m_dSumP + Base::m_sumP.transpose()*Base::m_dSumN)
94  - (Base::m_sumP.transpose()*Base::m_sumN)*m_d2SumW.transpose()
95  - (Base::m_dSumN.transpose()*Base::m_sumP + Base::m_dSumP.transpose()*Base::m_sumN)*Base::m_dSumW)
96  - Scalar(2.)*Base::m_sumW*Base::m_dSumW.transpose()*(Base::m_sumW*(Base::m_sumN.transpose()*Base::m_dSumP + Base::m_sumP.transpose()*Base::m_dSumN)
97  - (Base::m_sumP.transpose()*Base::m_sumN)*Base::m_dSumW));
98 
99  Matrix d2Deno = m_d2SumDotPP
100  - invSumW*invSumW*invSumW*invSumW*(
101  Base::m_sumW*Base::m_sumW*( Scalar(2.)*Base::m_sumW*(sumdSumPdSumP+sumd2SumPdSumP)
102  + Scalar(2.)*Base::m_dSumW.transpose()*(Base::m_sumP.transpose()*Base::m_dSumP)
103  - (Base::m_sumP.transpose()*Base::m_sumP)*m_d2SumW.transpose()
104  - Scalar(2.)*(Base::m_dSumP.transpose()*Base::m_sumP)*Base::m_dSumW)
105  - Scalar(2.)*Base::m_sumW*Base::m_dSumW.transpose()*(Scalar(2.)*Base::m_sumW*Base::m_sumP.transpose()*Base::m_dSumP
106  - (Base::m_sumP.transpose()*Base::m_sumP)*Base::m_dSumW));
107 
108  Scalar deno2 = Base::m_deno*Base::m_deno;
109 
110  m_d2Uq = Scalar(.5)/(deno2*deno2)*(deno2*( Base::m_dDeno.transpose()*Base::m_dNume
111  + Base::m_deno*d2Nume
112  - Base::m_dNume.transpose()*Base::m_dDeno
113  - Base::m_nume*d2Deno)
114  - Scalar(2.)*Base::m_deno*Base::m_dDeno.transpose()*(Base::m_deno*Base::m_dNume - Base::m_nume*Base::m_dDeno));
115 
116  for(int i=0; i<Dim; ++i)
117  {
118  m_d2Ul.template block<DerDim,DerDim>(0,i*DerDim) = invSumW*(
119  m_d2SumN.template block<DerDim,DerDim>(0,i*DerDim)
120  - Scalar(2.)*( m_d2Uq*Base::m_sumP[i]
121  + Base::m_dSumP.row(i).transpose()*Base::m_dUq
122  + Base::m_uq*m_d2SumP.template block<DerDim,DerDim>(0,i*DerDim)
123  + Base::m_dUq.transpose()*Base::m_dSumP.row(i))
124  - Base::m_ul[i]*m_d2SumW
125  - Base::m_dUl.row(i).transpose()*Base::m_dSumW
126  - Base::m_dSumW.transpose()*Base::m_dUl.row(i));
127  }
128 
129  Matrix sumdUldSumP = Matrix::Zero();
130  Matrix sumUld2SumP = Matrix::Zero();
131  Matrix sumd2UlsumP = Matrix::Zero();
132  Matrix sumdSumPdUl = Matrix::Zero();
133 
134  for(int i=0; i<Dim; ++i)
135  {
136  sumdUldSumP += Base::m_dUl.row(i).transpose()*Base::m_dSumP.row(i);
137  sumUld2SumP += Base::m_ul[i]*m_d2SumP.template block<DerDim,DerDim>(0,i*DerDim);
138  sumd2UlsumP += m_d2Ul.template block<DerDim,DerDim>(0,i*DerDim)*Base::m_sumP[i];
139  sumdSumPdUl += Base::m_dSumP.row(i).transpose()*Base::m_dUl.row(i);
140  }
141 
142  m_d2Uc = -invSumW*(
143  sumdUldSumP
144  + sumUld2SumP
145  + sumd2UlsumP
146  + sumdSumPdUl
147  + Base::m_dUq.transpose()*Base::m_dSumDotPP
148  + Base::m_uq*m_d2SumDotPP
149  + Base::m_dSumDotPP.transpose()*Base::m_dUq
150  + m_d2Uq*Base::m_sumDotPP
151  + Base::m_uc*m_d2SumW
152  + Base::m_dUc.transpose()*Base::m_dSumW
153  + Base::m_dSumW.transpose()*Base::m_dUc);
154  }
155 
156  return Base::m_eCurrentState;
157 }

159 template < class DataPoint, class _WFunctor, typename T>
160 typename MlsSphereFitDer<DataPoint, _WFunctor, T>::ScalarArray
161 MlsSphereFitDer<DataPoint, _WFunctor, T>::dPotential() const
162 {
163  // Compute the 1st order derivative of the scalar field s = uc + x^T ul + x^T x uq
164  // In a centered basis (x=0), we obtain:
165  // the scale derivative: d_t(s)(t,0) = d_t(uc)(t,0)
166  // the spatial derivative: d_x(s)(t,0) = d_x(uc)(t,0) + ul(t,0)
167  ScalarArray result = Base::m_dUc;
168 
169  if(Base::isSpaceDer())
170  result.template tail<Dim>() += Base::m_ul;
171 
172  return result;
173 }

184 template < class DataPoint, class _WFunctor, typename T>
185 typename MlsSphereFitDer<DataPoint, _WFunctor, T>::VectorArray
186 MlsSphereFitDer<DataPoint, _WFunctor, T>::dNormal() const
187 {
188  // Compute the 1st order derivative of the normal, which is the normalized gradient N = d_x(s) / |d_x(s)|
189  // We obtain:
190  // the scale derivative: d_t(N) = (I-N N^T) / |d_x(s)| d2_tx(s)
191  // the spatial derivative: d_x(N) = (I-N N^T) / |d_x(s)| d2_x2(s)
192  // Where in a centered basis (x=0), we have:
193  // d2_tx(s) = d2_tx(uc) + d_t(ul)
194  // d2_x2(s) = d2_x2(uc) + d_x(ul) + d_x(ul)^T + 2 uq I
195  VectorArray result = VectorArray::Zero();
196 
197  VectorType grad = Base::m_dUc.template tail<Dim>().transpose() + Base::m_ul;
198  Scalar gradNorm = grad.norm();
199 
200  if(Base::isScaleDer())
201  result.col(0) = m_d2Uc.template topRightCorner<1,Dim>().transpose() + Base::m_dUl.col(0);
202 
203  if(Base::isSpaceDer())
204  {
205  result.template rightCols<Dim>() = m_d2Uc.template bottomRightCorner<Dim,Dim>().transpose()
206  + Base::m_dUl.template rightCols<Dim>().transpose()
207  + Base::m_dUl.template rightCols<Dim>();
208  result.template rightCols<Dim>().diagonal().array() += Scalar(2.)*Base::m_uq;
209  }
210 
211  return result/gradNorm - grad*grad.transpose()/(gradNorm*gradNorm)*result;
212 }
