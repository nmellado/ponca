33 template < class DataPoint, class _WFunctor, typename T>
34 FIT_RESULT
35 CovarianceFitBase<DataPoint, _WFunctor, T>::finalize ()
36 {
37  // handle specific configurations
38  // With less than 3 neighbors the fitting is undefined
39  if(Base::finalize() != STABLE || Base::m_nbNeighbors < 3)
40  {
41  return Base::m_eCurrentState = UNDEFINED;
42  }
43 
44  // Center the covariance on the centroid
45  auto centroid = Base::barycenter();
46  m_cov = m_cov/Base::m_sumW - centroid * centroid.transpose();
47 
48 #ifdef __CUDACC__
49  m_solver.computeDirect(m_cov);
50 #else
51  m_solver.compute(m_cov);
52 #endif
53  Base::m_eCurrentState = ( m_solver.info() == Eigen::Success ? STABLE : UNDEFINED );
54 
55  return Base::m_eCurrentState;
56 }

96 template < class DataPoint, class _WFunctor, int DiffType, typename T>
97 FIT_RESULT
98 CovarianceFitDer<DataPoint, _WFunctor, DiffType, T>::finalize()
99 {
100  PONCA_MULTIARCH_STD_MATH(sqrt);
101 
102  Base::finalize();
103  // Test if base finalize end on a viable case (stable / unstable)
104  if (this->isReady())
105  {
106  VectorType cog = Base::barycenter();
107  MatrixType cogSq = cog * cog.transpose();
108  // \fixme Better use eigen here
109  for(int k=0; k<Base::NbDerivatives; ++k)
110  {
111  // Finalize the computation of dCov.
112  m_dCov[k] = m_dCov[k]
113  - cog * Base::m_dSumP.col(k).transpose()
114  - Base::m_dSumP.col(k) * cog.transpose()
115  + Base::m_dSumW[k] * cogSq;
116  }
117  }
118 
119  return Base::m_eCurrentState;
120 }
