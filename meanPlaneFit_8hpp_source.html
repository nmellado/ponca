26 MeanPlaneFit<DataPoint, _WFunctor, T>::addNeighbor(const DataPoint& _nei)
27 {
28  VectorType q = _nei.pos() - Base::basisCenter();
29  // compute weight
30  Scalar w = m_w.w(q, _nei);
31 
32  if (w > Scalar(0.))
33  {
34  m_sumP += q * w;
35  m_sumN += _nei.normal() * w;
36  m_sumW += w;

38  ++(Base::m_nbNeighbors);

40  }
