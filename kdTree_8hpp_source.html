84 template<class DataPoint>
85 bool KdTree<DataPoint>::valid() const
86 {
87  PONCA_DEBUG_ERROR;
88  return false;
89 
90  if (m_points.empty())
91  return m_nodes.empty() && m_indices.empty();
92 
93  if(m_nodes.empty() || m_indices.empty())
94  {
95  PONCA_DEBUG_ERROR;
96  return false;
97  }
98 
99  if(point_count() < index_count())
100  {
101  PONCA_DEBUG_ERROR;
102  return false;
103  }
104 
105  std::vector<bool> b(point_count(), false);
106  for(int idx : m_indices)
107  {
108  if(idx < 0 || point_count() <= idx || b[idx])
109  {
110  PONCA_DEBUG_ERROR;
111  return false;
112  }
113  b[idx] = true;
114  }
115 
116  for(int n=0;n<node_count();++n)
117  {
118  const KdTreeNode<Scalar>& node = m_nodes.operator[](n);
119  if(node.leaf)
120  {
121  if(index_count() <= node.start || index_count() < node.start+node.size)
122  {
123  PONCA_DEBUG_ERROR;
124  return false;
125  }
126  }
127  else
128  {
129  if(node.dim < 0 || 2 < node.dim)
130  {
131  PONCA_DEBUG_ERROR;
132  return false;
133  }
134  if(node_count() <= node.firstChildId || node_count() <= node.firstChildId+1u)
135  {
136  PONCA_DEBUG_ERROR;
137  return false;
138  }
139  }
140  }
141 
142  return true;
143 }

185 template<class DataPoint>
186 void KdTree<DataPoint>::build_rec(int node_id, int start, int end, int level)
187 {
188  KdTreeNode<Scalar>& node = m_nodes[node_id];
189  Aabb aabb;
190  for(int i=start; i<end; ++i)
191  aabb.extend(m_points[m_indices[i]].pos());
192 
193  int dim;
194  (Scalar(0.5) * (aabb.max() - aabb.min())).maxCoeff(&dim);
195  node.dim = dim;
196  node.splitValue = aabb.center()(dim);
197 
198  int midId = this->partition(start, end, dim, node.splitValue);
199  node.firstChildId = m_nodes.size();
200 
201  {
202  KdTreeNode<Scalar> n;
203  n.size = 0;
204  m_nodes.push_back(n);
205  m_nodes.push_back(n);
206  }
207  {
208  // left child
209  int childId = m_nodes[node_id].firstChildId;
210  KdTreeNode<Scalar>& child = m_nodes[childId];
211  if(midId-start <= m_min_cell_size || level >= PCA_KDTREE_MAX_DEPTH)
212  {
213  child.leaf = 1;
214  child.start = start;
215  child.size = midId-start;
216  }
217  else
218  {
219  child.leaf = 0;
220  this->build_rec(childId, start, midId, level+1);
221  }
222  }
223  {
224  // right child
225  int childId = m_nodes[node_id].firstChildId+1;
226  KdTreeNode<Scalar>& child = m_nodes[childId];
227  if(end-midId <= m_min_cell_size || level >= PCA_KDTREE_MAX_DEPTH)
228  {
229  child.leaf = 1;
230  child.start = midId;
231  child.size = end-midId;
232  }
233  else
234  {
235  child.leaf = 0;
236  this->build_rec(childId, midId, end, level+1);
237  }
238  }
239 }

241 template<class DataPoint>
242 int KdTree<DataPoint>::partition(int start, int end, int dim, Scalar value)
243 {
244  const auto& points = m_points;
245  auto& indices = m_indices;
246 
247  auto it = std::partition(indices.begin()+start, indices.begin()+end, [&](int i)
248  {
249  return points[i].pos()[dim] < value;
250  });
251 
252  auto distance = std::distance(m_indices.begin(), it);
253 
254  return static_cast<int>(distance);
255 }
