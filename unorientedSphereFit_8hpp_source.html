101 template < class DataPoint, class _WFunctor, typename T>
102 FIT_RESULT
103 UnorientedSphereFit<DataPoint, _WFunctor, T>::finalize ()
104 {
105  PONCA_MULTIARCH_STD_MATH(sqrt);
106 
107  // 1. finalize sphere fitting
108 
109  // handle specific configurations
110  // With less than 3 neighbors the fitting is undefined
111  if(m_sumW == Scalar(0.) || Base::m_nbNeighbors < 3)
112  {
113  Base::m_ul.setZero();
114  Base::m_uc = 0;
115  Base::m_uq = 0;
116  Base::m_isNormalized = false;
117  Base::m_eCurrentState = UNDEFINED;
118  return Base::m_eCurrentState;
119  }
120 
121  Scalar invSumW = Scalar(1.) / m_sumW;
122 
123 
124  MatrixBB Q;
125  Q.template topLeftCorner<Dim,Dim>().setIdentity();
126  Q.col(Dim).template head<Dim>() = m_sumP * invSumW;
127  Q.row(Dim).template head<Dim>() = m_sumP * invSumW;
128  Q(Dim,Dim) = m_sumDotPP * invSumW;
129  m_matA *= invSumW;
130 
131  MatrixBB M = Q.inverse() * m_matA;
132  Eigen::EigenSolver<MatrixBB> eig(M);
133  VectorB eivals = eig.eigenvalues().real();
134  int maxId = 0;
135  eivals.maxCoeff(&maxId);
136 
137  VectorB eivec = eig.eigenvectors().col(maxId).real();
138 
139  // integrate
140  Base::m_ul = eivec.template head<Dim>();
141  Base::m_uq = Scalar(0.5) * eivec(Dim);
142  Base::m_uc = -invSumW * (Base::m_ul.dot(m_sumP) + m_sumDotPP * Base::m_uq);
143 
144  Base::m_isNormalized = false;
145 
146  if(Base::m_nbNeighbors < 6)
147  {
148  Base::m_eCurrentState = UNSTABLE;
149  }
150  else
151  {
152  Base::m_eCurrentState = STABLE;
153  }
154 
155  return Base::m_eCurrentState;
156 }

228  // Test if base finalize end on a viable case (stable / unstable)
229  if (this->isReady())

232  Scalar invSumW = Scalar(1.)/Base::m_sumW;
233 
234  Scalar nume = Base::m_sumDotPN - invSumW*Base::m_sumP.dot(Base::m_sumN);
235  Scalar deno = Base::m_sumDotPP - invSumW*Base::m_sumP.dot(Base::m_sumP);
236 
237  ScalarArray dNume = m_dSumDotPN - invSumW*invSumW * ( Base::m_sumW * (
238  Base::m_sumN.transpose() * m_dSumP +
239  Base::m_sumP.transpose() * m_dSumN )
240  - m_dSumW*Base::m_sumP.dot(Base::m_sumN) );
241  ScalarArray dDeno = m_dSumDotPP - invSumW*invSumW*( Scalar(2.)*Base::m_sumW * Base::m_sumP.transpose()*m_dSumP
242  - m_dSumW*Base::m_sumP.dot(Base::m_sumP) );
243 
244  m_dUq = Scalar(.5) * (deno * dNume - dDeno * nume)/(deno*deno);
245  m_dUl = invSumW*((m_dSumN - Scalar(2.)*(m_dSumP*Base::m_uq+Base::m_sumP*m_dUq)) - Base::m_ul*m_dSumW);
246  m_dUc = -invSumW*( Base::m_sumP.transpose() * m_dUl +
247  Base::m_sumDotPP * m_dUq + Base::m_ul.transpose() * m_dSumP +
248  Base::m_uq*m_dSumDotPP + m_dSumW*Base::m_uc);
249  }

256 template < class DataPoint, class _WFunctor, typename T, int Type>
257 bool
258 OrientedSphereDer<DataPoint, _WFunctor, T, Type>::applyPrattNorm()
259 {
260  if(Base::isNormalized())
261  return false; //need original parameters without Pratt Normalization
262 
263 
264  PONCA_MULTIARCH_STD_MATH(sqrt);
265  Scalar pn2 = Base::prattNorm2();
266  Scalar pn = sqrt(pn2);
267 
268  ScalarArray dpn2 = dprattNorm2();
269  ScalarArray factor = Scalar(0.5) * dpn2 / pn;
270 
271  m_dUc = ( m_dUc * pn - Base::m_uc * factor ) / pn2;
272  m_dUl = ( m_dUl * pn - Base::m_ul * factor ) / pn2;
273  m_dUq = ( m_dUq * pn - Base::m_uq * factor ) / pn2;
274 
275  Base::applyPrattNorm();
276  return true;
277 }
