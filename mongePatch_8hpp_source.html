33  const Scalar& u = *(local.data()+1);

35 
36  Eigen::Matrix<Scalar, 6, 1 > p;
37  p << u*u, v*v, u*v, u, v, 1;
38  m_A += w*p*p.transpose();
39  m_b += w*h*p;
40 
41  return true;
42  }
43  }
44  return false;
